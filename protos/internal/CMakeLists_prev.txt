INCLUDE(GRPCGenerateCPP)

find_package(Protobuf REQUIRED)

find_program(GRPC_CPP_PLUGIN grpc_cpp_plugin)
if(NOT GRPC_CPP_PLUGIN)
  message(FATAL_ERROR "grpc_cpp_plugin not found")
endif()

set(INTERNAL_PROTO_FILES
    query.proto
    diffusion_service.proto
)


set(GENERATED_GRPC_SRCS "")
set(GENERATED_GRPC_HDRS "")

foreach(proto ${PROTO_FILES})
  get_filename_component(proto_name ${proto} NAME_WE)
  list(APPEND GENERATED_GRPC_SRCS
    ${CMAKE_CURRENT_BINARY_DIR}/${proto_name}.grpc.pb.cc)
  list(APPEND GENERATED_GRPC_HDRS
    ${CMAKE_CURRENT_BINARY_DIR}/${proto_name}.grpc.pb.h)
endforeach()

add_custom_command(
  OUTPUT ${GENERATED_GRPC_SRCS} ${GENERATED_GRPC_HDRS}
  COMMAND ${Protobuf_PROTOC_EXECUTABLE}
  ARGS
    --grpc_out=${CMAKE_CURRENT_BINARY_DIR}
    --cpp_out=${CMAKE_CURRENT_BINARY_DIR}
    --plugin=protoc-gen-grpc=${GRPC_CPP_PLUGIN}
    -I ${CMAKE_CURRENT_SOURCE_DIR}
    ${PROTO_FILES}
  DEPENDS ${PROTO_FILES}
)

add_library(infaas-protos-internal STATIC
  ${PROTO_FILES}
  ${GENERATED_GRPC_SRCS}
  ${GENERATED_GRPC_HDRS}
)


PROTOBUF_GENERATE_CPP(PROTO_SRCS PROTO_HDRS
  query.proto
  sys_monitor.proto
  infaas_request_status.proto
  ${INTERNAL_PROTO_FILES}
)

PROTOBUF_GENERATE_GRPC_CPP(PROTO_GRPC_SRCS PROTO_GRPC_HDRS
  query.proto
  sys_monitor.proto
  diffusion_service.proto
)


protobuf_generate(
    LANGUAGE grpc
    TARGET infaas-protos-internal
    PROTOS ${PROTO_FILES}
)

add_library(infaas-protos-internal
  ${PROTO_SRCS}
  ${PROTO_GRPC_SRCS}
)

target_include_directories(infaas-protos-internal PUBLIC
  ${CMAKE_CURRENT_BINARY_DIR}
  ${CMAKE_CURRENT_SOURCE_DIR}
  ${Protobuf_INCLUDE_DIRS}
  ${PROJECT_BINARY_DIR}/protos/
)

target_link_libraries(infaas-protos-internal
    protobuf::libprotobuf
    grpc++
    grpc
)
